import React,{useState,useEffect} from "react"
function FetchD(){
    const [data,setData]=useState([])
    useEffect(()=>{
        fetch("https://jsonplaceholder.typicode.com/todos").then((res)=>res.json()).then((myda)=>setData(myda))
    },[])
  return(
    <div>
        {
            data.map((item)=><li>{item.id}</li>)
        }
    </div>
  )
}

export default FetchD;
